<script>
function loadStory(storyId) {
fetch(`dados/historia${storyId}.html`)
.then(response => response.text())
.then(htmlContent => {
const parser = new DOMParser();
const doc = parser.parseFromString(htmlContent, 'text/html');
const paragraphs = doc.querySelectorAll('p');
let processedHTML = '';
paragraphs.forEach(p => {
let text = p.innerHTML.trim();
if (!text) return; // Ignora linhas vazias

// Detecta se é uma linha de fala contendo a tag de negrito do emissor
if (text.includes('class="text-[#E30613]"') || text.includes("class='text-[#E30613]'")) {
const boldTag = p.querySelector('b');
if (boldTag) {
const senderName = boldTag.textContent.replace(':', '').trim();
// Remove a tag <b> do texto para deixar apenas o diálogo limpo
boldTag.remove();
let dialogue = p.innerHTML.trim();

// Determina o lado e cores do avatar com base no emissor (Fusão de UI)
let isStaff = senderName.includes('BRUNO') || senderName.includes('PROMOTOR') || senderName.includes('EDY') || senderName.includes('Efy');
let sideClass = isStaff ? 'left' : 'right';
let avatarColor = isStaff ? '%23e30613' : '%2300b2ff';

processedHTML += `
<div class="chat-msg ${sideClass}">
<div class="chat-avatar">
<img src="../avatares/${senderName.toLowerCase().split(' ')[0]}.png" onerror="this.src='data:image/svg+xml;utf8,<svg xmlns=\'http://www.w3.org/2000/svg\' viewBox=\'0 0 24 24\' fill=\'${avatarColor}\'><circle cx=\'12\' cy=\'8\' r=\'4\'/><path d=\'M12 14c-6.1 0-12 2.5-12 6v2h24v-2c0-3.5-5.9-6-12-6z\'/></svg>';">
</div>
<div style="width: 100%;">
<span class="sender-tag" style="text-align: ${isStaff ? 'left' : 'right'};">${senderName}</span>
<div class="bubble">${dialogue}</div>
</div>
</div>
`;
return;
}
}

// Se for uma anotação de contexto estrutural, renderiza dentro de um painel explicativo sem perder a informação
if (text.startsWith('[') && text.endsWith(']')) {
processedHTML += `
<div style="text-align: center; margin: 10px 0;">
<span style="background: var(--border-default); color: var(--text-secondary); font-size: 10px; font-weight: 700; padding: 4px 10px; border-radius: 12px; text-transform: uppercase; letter-spacing: 0.5px;">
${text.replace('[', '').replace(']', '')}
</span>
</div>
`;
} else {
// Títulos das seções iniciais ou notas narrativas
processedHTML += `
<div class="cyber-panel" style="padding: 10px; margin-bottom: 2px; border-left: 3px solid var(--neon-cyan);">
<p class="panel-text" style="font-size: 12px; font-weight: 600;">${text}</p>
</div>
`;
}
});

// Injeta as bolhas estruturadas na área de chat
document.getElementById('chat-container').innerHTML = processedHTML;

// Atualiza o estado visual dos botões das abas superiores
document.querySelectorAll('[id^=b]').forEach((btn, idx) => {
if ((idx + 1) === storyId) {
btn.style.backgroundColor = '#E30613';
btn.style.color = '#ffffff';
} else {
btn.style.backgroundColor = 'var(--bg-card)';
btn.style.color = 'var(--text-secondary)';
btn.style.border = '1px solid var(--border-default)';
}
});

// Atualiza o botão de rodapé dinâmico para avançar
const navContainer = document.getElementById('story-navigation');
if (storyId < 4) {
navContainer.innerHTML = `
<button onclick="loadStory(${storyId + 1})" style="width: 100%; text-align: center; padding: 14px; background-color: #E30613; color: #ffffff; border-radius: 10px; font-size: 13px; font-weight: 700; border: none; cursor: pointer;">
Próxima História →
</button>
`;
} else {
// Redireciona para o novo hub central de Conclusões
navContainer.innerHTML = `
<div style="display: grid; grid-template-columns: 1fr; gap: 10px;">
<a href="../conclusoes/index.html" style="padding: 14px; background-color: #E30613; color: #fff; border-radius: 10px; text-align: center; font-size: 13px; font-weight: 700; border: none; cursor: pointer; text-decoration: none;">
Ver Conclusões
</a>
</div>
`;
}
}).catch(err => {
document.getElementById('chat-container').innerHTML = `<p class="panel-text" style="color:var(--neon-red);">Erro ao carregar história: ${err}</p>`;
});
}

// Carrega a primeira história automaticamente na inicialização
document.addEventListener("DOMContentLoaded", () => {
loadStory(1);
});
</script>
